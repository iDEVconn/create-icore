- [ ] **Step 2: Controller**

```ts
@Controller()
export class NotesController {
  constructor(@Inject('DBStrategy') private readonly db: DBStrategy) {}

  @MessagePattern('notes.list')
  async list(@Payload() p: ListNotesOptions): Promise<{ items: Note[]; total: number }> {
    const where = p.ownerId
      ? [{ field: 'ownerId', op: '==' as const, value: p.ownerId }]
      : undefined;
    const all = await this.db.list<Note>('notes', {
      where,
      orderBy: { field: 'createdAt', direction: 'desc' },
    });
    return {
      items: all.slice(p.offset, p.offset + p.limit).map((d) => d.data),
      total: all.length,
    };
  }

  @MessagePattern('notes.get')
  async get(@Payload() p: { id: string }): Promise<Note | null> {
    const doc = await this.db.get<Note>('notes', p.id);
    return doc?.data ?? null;
  }

  @MessagePattern('notes.create')
  async create(@Payload() p: { ownerId: string; title: string; body: string }): Promise<Note> {
    const id = randomUUID();
    const now = new Date().toISOString();
    const note: Note = {
      id,
      ownerId: p.ownerId,
      title: p.title,
      body: p.body,
      createdAt: now,
      updatedAt: now,
    };
    await this.db.set('notes', id, note);
    return note;
  }

  @MessagePattern('notes.update')
  async update(
    @Payload() p: { id: string; patch: Partial<Pick<Note, 'title' | 'body'>> },
  ): Promise<Note> {
    await this.db.update<Note>('notes', p.id, { ...p.patch, updatedAt: new Date().toISOString() });
    const doc = await this.db.get<Note>('notes', p.id);
    if (!doc) throw new Error('note_missing_after_update');
    return doc.data;
  }

  @MessagePattern('notes.delete')
  delete(@Payload() p: { id: string }): Promise<void> {
    return this.db.delete('notes', p.id);
  }
}
```

- [ ] **Step 3: Gateway controller**

```ts
@ApiTags('notes')
@Controller('notes')
export class NotesController {
  constructor(
    private readonly notes: NotesClientService,
    private readonly abilities: AbilityFactory,
  ) {}

  @Get()
  async list(
    @Req() req: AuthedRequest,
    @Query('limit') limit = '20',
    @Query('offset') offset = '0',
  ) {
    const ability = this.abilities.forUser(req.user);
    const ownerId = ability.can('manage', 'Note') ? null : req.user.id;
    return this.notes.list({ ownerId, limit: Number(limit), offset: Number(offset) });
  }

  @Get(':id')
  async get(@Req() req: AuthedRequest, @Param('id') id: string) {
    const note = await this.notes.get(id);
    if (!note) throw new NotFoundException();
    const ability = this.abilities.forUser(req.user);
    if (!ability.can('read', subject('Note', note))) throw new ForbiddenException();
    return note;
  }

  @Post()
  @CheckAbility('create', 'Note')
  create(@Req() req: AuthedRequest, @Body() body: { title: string; body: string }) {
    return this.notes.create({ ownerId: req.user.id, title: body.title, body: body.body });
  }

  @Patch(':id')
  async update(
    @Req() req: AuthedRequest,
    @Param('id') id: string,
    @Body() patch: { title?: string; body?: string },
  ) {
    const existing = await this.notes.get(id);
    if (!existing) throw new NotFoundException();
    const ability = this.abilities.forUser(req.user);
    if (!ability.can('update', subject('Note', existing))) throw new ForbiddenException();
    return this.notes.update(id, patch);
  }

  @Delete(':id')
  @HttpCode(204)
  async delete(@Req() req: AuthedRequest, @Param('id') id: string) {
    const existing = await this.notes.get(id);
    if (!existing) throw new NotFoundException();
    const ability = this.abilities.forUser(req.user);
    if (!ability.can('delete', subject('Note', existing))) throw new ForbiddenException();
    await this.notes.delete(id);
  }
}
```
