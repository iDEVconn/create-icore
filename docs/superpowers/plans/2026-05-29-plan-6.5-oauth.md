```ts
async startOAuth(provider: OAuthProvider, callbackUrl: string): Promise<OAuthStartResult> {
  const { data, error } = await this.client.auth.signInWithOAuth({
    provider,
    options: { redirectTo: callbackUrl, skipBrowserRedirect: true },
  });
  if (error || !data?.url) throw new Error(error?.message ?? 'oauth_start_failed');
  const url = new URL(data.url);
  const state = url.searchParams.get('state') ?? randomUUID();
  return { redirectUrl: data.url, state };
}

async completeOAuth(_provider: OAuthProvider, code: string, _state: string): Promise<AuthSession> {
  const { data, error } = await this.client.auth.exchangeCodeForSession(code);
  if (error || !data.session) throw new Error(error?.message ?? 'oauth_complete_failed');
  return this.toSession(data.session);
}
```

- [ ] **Step 2: Strategy methods**

```ts
async startOAuth(provider: OAuthProvider, callbackUrl: string): Promise<OAuthStartResult> {
  const state = randomUUID();
  this.pendingStates.set(state, provider);
  const clientId = this.cfg.oauth[provider].clientId;
  const scopes = provider === 'google' ? 'openid email profile' : 'read:user user:email';
  const base =
    provider === 'google'
      ? 'https://accounts.google.com/o/oauth2/v2/auth'
      : 'https://github.com/login/oauth/authorize';
  const url = new URL(base);
  url.searchParams.set('client_id', clientId);
  url.searchParams.set('redirect_uri', callbackUrl);
  url.searchParams.set('response_type', 'code');
  url.searchParams.set('scope', scopes);
  url.searchParams.set('state', state);
  return { redirectUrl: url.toString(), state };
}

async completeOAuth(provider: OAuthProvider, code: string, state: string): Promise<AuthSession> {
  if (!this.pendingStates.delete(state)) throw new Error('invalid_oauth_state');
  const { clientId, clientSecret } = this.cfg.oauth[provider];
  // Exchange code → idToken / accessToken via provider's token endpoint
  const tokenRes = await this.tokenClient.exchange(provider, { code, clientId, clientSecret });
  // Sign into Firebase with the provider idToken/accessToken
  const postBody =
    provider === 'google'
      ? `id_token=${tokenRes.idToken}&providerId=google.com`
      : `access_token=${tokenRes.accessToken}&providerId=github.com`;
  const res = await this.identityToolkit.signInWithIdp({
    requestUri: 'http://localhost', // origin doesn't matter for server flow
    postBody,
  });
  return {
    accessToken: res.idToken,
    refreshToken: res.refreshToken,
    expiresIn: Number(res.expiresIn),
    user: { id: res.localId, email: res.email },
  };
}
```

New helper `tokenClient` — small `OAuthTokenClient` interface with two endpoints (`google` → `https://oauth2.googleapis.com/token`, `github` → `https://github.com/login/oauth/access_token`). Mock impl for tests; real impl uses `fetch`.

- [ ] **Step 3: Gateway routes**

```ts
@Public()
@Get('oauth/:provider')
async oauthStart(@Param('provider') provider: OAuthProvider, @Res() res: Response) {
  const origin = this.cfg.get<string>('CLIENT_ORIGIN') ?? 'http://localhost:4200';
  const callbackUrl = `${origin}/api/auth/oauth/${provider}/callback`;
  const { redirectUrl, state } = await this.authClient.startOAuth(provider, callbackUrl);
  res.cookie('oauth_state', state, {
    httpOnly: true,
    secure: this.cfg.get('NODE_ENV') === 'production',
    sameSite: 'lax',
    maxAge: 10 * 60 * 1000,
  });
  return res.redirect(redirectUrl);
}

@Public()
@Get('oauth/:provider/callback')
async oauthCallback(
  @Param('provider') provider: OAuthProvider,
  @Query('code') code: string,
  @Query('state') state: string,
  @Req() req: Request,
  @Res() res: Response,
) {
  const cookieState = req.cookies?.['oauth_state'];
  if (!cookieState || cookieState !== state) {
    throw new UnauthorizedException('oauth_state_mismatch');
  }
  const session = await this.authClient.completeOAuth(provider, code, state);
  res.clearCookie('oauth_state');
  const origin = this.cfg.get<string>('CLIENT_ORIGIN') ?? 'http://localhost:4200';
  const fragment = new URLSearchParams({
    accessToken: session.accessToken,
    refreshToken: session.refreshToken,
    userId: session.user.id,
    email: session.user.email,
  });
  return res.redirect(`${origin}/auth/oauth/callback#${fragment.toString()}`);
}
```

Both routes are `@Public()` + inside the existing `auth-burst` throttle.
